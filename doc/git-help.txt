# Installer git
sudo apt install git

# Afficher la version de Git ( ectuelle 2.43.0 )
git -v

# Initialiser un environnement github
git init

# Ajouter des fichiers au projet
git add -A gère tous les changements
ou
git add --all
git add . gère les nouveaux fichiers et les modifications, sans les suppressions
git add -u gère les modifications et les suppressions, sans les nouveaux fichiers.
ou
git add --update

# Annuler les changement
git reset

# Recuperer les modifs faites sur le seveur distant
git fetch

# Examiner la difference 
git diff
ou
git log

# creer une nouvelle branche
git checkout -b developer
ou
git branch developer

# Lister toutes les branches
git branch -a

# Changer de branche
git switch main

# Importer du code
git clone https://github.com/repo.git

# Pousser du code pour la premiere fois
git push -u origin develop
ou
git push --set-upstream origin develop

# Fusionner la partie develop avec la partie main
git checkout master
puis
git merge develop

# Supprimer une branche github distante
git push -d origin develop
ou
git push --delete origin develop

# Supprimer une branche github locale
git branch -d developer
ou
git branch --delete developer

# Scenario de fusion de deux branches : 
#la branche main contient la version stable et en production de l'application.
# main ou bien release ou production.

# Etape 1 : Verifier ou en est le code et sur quel branche ou se trouve.
git status
git branch

# Etape 2 : S'assurer que la branch main est a jour.
git checkout main
git pull origin main

# Etape 2 : S'assurer que la branch develop est a jour.
git checkout develop
git pull origin develop

# Etape 3 : retourner sur la branch main et fusionner avec develop
git checkout main
git merge develop



### Les remotes
# Ajouter un depot distant a un repo.
git remote add myremote myrepo

# Ajouter un depot distant au repo local
git remote add myremote .

# Afficher les remote
git remote -v

# Pousser les modifs local sur la branche correspondant sur la remote.
git push remote branche

# Recuperer les modifs sur la branch remote et fusionner dans la branche local
git pull remote branch
