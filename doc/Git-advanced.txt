#Etape 1 : retrouver le commit ayant la bonne version du fichier
git log -- chemin/du/fichier_supprime.txt

# Etape 2 :  Lancer la restauration d'apres le commit de suppression -1
git checkout abcdef1^ -- chemin/du/fichier_supprime.txt
# Ou
git checkout abcdef1~1 -- chemin/du/fichier_supprime.txt

#ou restaurer depuis le commit de suppression si il existe directement.
git checkout commit_de_suppression -- checmin/du/fichier.txt

# Etape 3 : relancer le ficheir dans le tracking
git add file.txt
git commit -m "Restauration file.txt"
